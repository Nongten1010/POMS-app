5. Data Dictionary: chat_attachments

หน้า Chat ปัจจุบันแสดงชื่อไฟล์แนบรวมในข้อความ แต่ฐานข้อมูลควรแยกไฟล์แนบเพื่อเก็บ path, ขนาดไฟล์ และชนิดไฟล์ได้ชัดเจน

Column Type Key Required คำอธิบาย
idbigint identityPKYesรหัสไฟล์แนบ
message_idbigintFKYesข้อความที่แนบไฟล์นี้
file_namenvarchar(255)Yesชื่อไฟล์ต้นฉบับที่แสดงบนหน้า Chat
file_pathnvarchar(500)Yesตำแหน่งไฟล์ใน storage หรือ path ที่ backend ใช้ดาวน์โหลด
mime_typevarchar(100)Noชนิดไฟล์ เช่น application/pdf หรือ image/png
file_size_bytesbigintNoขนาดไฟล์ ใช้ตรวจสอบข้อจำกัดการแนบไฟล์
uploaded_atdatetime2Yesเวลาที่อัปโหลดไฟล์

6. SQL Server DDL แบบพอใช้

ตัวอย่างนี้ตั้งใจให้เป็น schema เริ่มต้นสำหรับทำ backend ต่อ ยังไม่บังคับ FK ไปหา users หรือ factories เพราะหน้า Chat ปัจจุบันยังเป็น mock data ใน frontend

CREATE TABLE chat_threads (
  id BIGINT IDENTITY(1,1) NOT NULL PRIMARY KEY,
  title NVARCHAR(300) NOT NULL,
  status VARCHAR(20) NOT NULL,
  category NVARCHAR(100) NOT NULL,
  operator_name NVARCHAR(255) NULL,
  operator_factory_id BIGINT NULL,
  created_by_user_id BIGINT NULL,
  assigned_to_user_id BIGINT NULL,
  created_at DATETIME2 NOT NULL CONSTRAINT df_chat_threads_created_at DEFAULT SYSUTCDATETIME(),
  updated_at DATETIME2 NOT NULL CONSTRAINT df_chat_threads_updated_at DEFAULT SYSUTCDATETIME(),
  closed_at DATETIME2 NULL,
  CONSTRAINT ck_chat_threads_status CHECK (status IN ('waiting', 'inProgress', 'closed'))
);

CREATE TABLE chat_messages (
  id BIGINT IDENTITY(1,1) NOT NULL PRIMARY KEY,
  thread_id BIGINT NOT NULL,
  sender_role VARCHAR(20) NOT NULL,
  sender_name NVARCHAR(150) NOT NULL,
  sender_user_id BIGINT NULL,
  message_text NVARCHAR(MAX) NOT NULL,
  created_at DATETIME2 NOT NULL CONSTRAINT df_chat_messages_created_at DEFAULT SYSUTCDATETIME(),
  is_internal_note BIT NOT NULL CONSTRAINT df_chat_messages_internal DEFAULT 0,
  CONSTRAINT fk_chat_messages_thread FOREIGN KEY (thread_id) REFERENCES chat_threads(id),
  CONSTRAINT ck_chat_messages_sender_role CHECK (sender_role IN ('operator', 'officer'))
);

CREATE TABLE chat_attachments (
  id BIGINT IDENTITY(1,1) NOT NULL PRIMARY KEY,
  message_id BIGINT NOT NULL,
  file_name NVARCHAR(255) NOT NULL,
  file_path NVARCHAR(500) NOT NULL,
  mime_type VARCHAR(100) NULL,
  file_size_bytes BIGINT NULL,
  uploaded_at DATETIME2 NOT NULL CONSTRAINT df_chat_attachments_uploaded_at DEFAULT SYSUTCDATETIME(),
  CONSTRAINT fk_chat_attachments_message FOREIGN KEY (message_id) REFERENCES chat_messages(id)
);

CREATE INDEX ix_chat_threads_status_updated_at ON chat_threads(status, updated_at DESC);
CREATE INDEX ix_chat_threads_category ON chat_threads(category);
CREATE INDEX ix_chat_messages_thread_created_at ON chat_messages(thread_id, created_at);
CREATE INDEX ix_chat_attachments_message_id ON chat_attachments(message_id);
หมายเหตุ: ถ้าต้องทำ production จริง ควรเพิ่ม endpoint, validation, upload storage policy, permission chat:ask/chat:answer และ foreign key ไปยังตารางผู้ใช้/โรงงานตาม schema จริงของ backend
