1. ตารางหลักที่เสนอ

ตาราง หน้าที่ หมายเหตุสำคัญ
bod_cod_deviation_reports หัวรายงาน 1 ฉบับ เก็บข้อมูลโรงงาน ห้องแล็บ เครื่องมือ ผู้รายงาน และสถานะรวม ใช้สร้าง PDF/HTML แบบรายงานผลการตรวจสอบความคลาดเคลื่อน
bod_cod_deviation_measurements แถวผลตรวจวัดของ parameter ที่เลือก 1 รายการ: วันที่ เวลา ค่าเครื่องมือ M ค่าแล็บ T ค่า E และค่าตามประกาศ เลือกได้ครั้งละ 1 รายการเท่านั้น (BOD หรือ COD) และคำนวณ E = M - T ตอนบันทึกหรือ generated column ก็ได้
bod_cod_deviation_attachments ไฟล์แนบ 3 ประเภท: ภาพเก็บตัวอย่าง, ภาพหน้าเครื่อง, รายงานผลแล็บ จำกัดชนิดไฟล์ JPG/PNG/PDF และขนาดไม่เกิน 2 MB ต่อไฟล์
bod_cod_deviation_reviews ผลตรวจแบบแจ้งผล: ความถูกต้องของแบบรายงาน, ผลค่าความคลาดเคลื่อน, เหตุผล, parameter ที่พิจารณา ใช้ข้อมูลชุดเดียว render ได้ทั้งส่วนกลางและภูมิภาค
bod_cod_approval_steps ขั้นอนุมัติรายคน แยกตามบทบาท: ผู้ตรวจสอบ, ผู้ทบทวน, ผู้อนุมัติ ส่วนกลางมี 3 step, ส่วนภูมิภาคมี 2 step แต่โครงตารางเดียวกัน
bod_cod_approval_events ประวัติทุก action: submit, approve, request_revision, edit_after_approval, resubmit จำเป็นเพราะผู้อนุมัติแต่ละคนแก้ไขผลของตนเองย้อนหลังได้

ตาราง ฟิลด์

3. สถานะรายงาน

DRAFTโรงงานกรอกข้อมูลและแนบไฟล์
SUBMITTEDส่งรายงานแล้ว ระบบเลือก CENTRAL/REGIONAL จากจังหวัด
UNDER_REVIEWรอผู้ตรวจสอบบันทึกผลตรวจ
REVIEWEDผู้ตรวจสอบอนุมัติแล้ว รอขั้นถัดไป
WAITING_APPROVALส่วนกลางรอผู้ทบทวน/ผู้อนุมัติ ส่วนภูมิภาครอผู้อนุมัติ
APPROVEDทุก step อนุมัติครบ ออกแบบแจ้งผลได้
REVISION_REQUESTEDมีคนขอแก้ไข โรงงานแก้และส่งใหม่

4. Step อนุมัติแยกตามพื้นที่

ส่วนกลาง - ภาคกลาง

1. ผู้ตรวจสอบ (INSPECTOR)

2. ผู้ทบทวน (REVIEWER)

3. ผู้อนุมัติ (APPROVER)

ส่วนภูมิภาค - จังหวัดอื่น

1. ผู้ตรวจสอบ (INSPECTOR)

2. ผู้อนุมัติ (APPROVER)

แก้ไขย้อนหลัง

คนที่อนุมัติแล้วแก้ข้อมูลของ step ตนเองได้ โดยเพิ่ม revision_no และบันทึก event edit_after_approval ไม่ลบประวัติเดิม

5. SQL draft

CREATE TABLE bod_cod_deviation_reports (
  id BIGINT IDENTITY(1,1) PRIMARY KEY,
  report_no VARCHAR(40) NOT NULL,
  report_round INT NOT NULL,
  report_year INT NOT NULL,
  factory_id BIGINT NULL,
  factory_name NVARCHAR(255) NOT NULL,
  factory_registration_no NVARCHAR(80) NOT NULL,
  business_activity NVARCHAR(255) NULL,
  address NVARCHAR(1000) NULL,
  province_name NVARCHAR(120) NOT NULL,
  approval_track VARCHAR(16) NOT NULL, -- CENTRAL | REGIONAL
  wastewater_flow_m3_per_hour DECIMAL(12,2) NULL,
  sampler_name NVARCHAR(255) NULL,
  officer_registration_no NVARCHAR(80) NULL,
  laboratory_name NVARCHAR(255) NULL,
  laboratory_registration_no NVARCHAR(80) NULL,
  lab_report_no NVARCHAR(120) NULL,
  analysis_method NVARCHAR(255) NULL,
  device_brand NVARCHAR(120) NULL,
  device_model NVARCHAR(120) NULL,
  device_serial_no NVARCHAR(120) NULL,
  reporter_name NVARCHAR(255) NULL,
  reporter_position NVARCHAR(255) NULL,
  status VARCHAR(32) NOT NULL DEFAULT 'DRAFT',
  submitted_at DATETIME2 NULL,
  created_by BIGINT NOT NULL,
  updated_by BIGINT NULL,
  created_at DATETIME2 NOT NULL DEFAULT SYSDATETIME(),
  updated_at DATETIME2 NOT NULL DEFAULT SYSDATETIME(),
  deleted_at DATETIME2 NULL
);

CREATE TABLE bod_cod_deviation_measurements (
  id BIGINT IDENTITY(1,1) PRIMARY KEY,
  report_id BIGINT NOT NULL,
  parameter_code VARCHAR(8) NOT NULL, -- BOD | COD
  sample_date DATE NOT NULL,
  sample_time TIME(0) NOT NULL,
  device_value_mg_l DECIMAL(12,3) NOT NULL,
  lab_value_mg_l DECIMAL(12,3) NOT NULL,
  deviation_value_mg_l AS (device_value_mg_l - lab_value_mg_l),
  standard_deviation_mg_l DECIMAL(12,3) NULL,
  is_within_standard BIT NULL,
  sort_order INT NOT NULL DEFAULT 1,
  CONSTRAINT fk_bodcod_measure_report FOREIGN KEY (report_id)
    REFERENCES bod_cod_deviation_reports(id)
);

CREATE TABLE bod_cod_deviation_reviews (
  report_id BIGINT NOT NULL PRIMARY KEY,
  selected_parameter_code VARCHAR(8) NOT NULL, -- BOD | COD
  form_check_result VARCHAR(16) NULL, -- CORRECT | INCORRECT
  form_check_reason NVARCHAR(MAX) NULL,
  deviation_check_result VARCHAR(20) NULL, -- COMPLIANT | NON_COMPLIANT
  deviation_check_reason NVARCHAR(MAX) NULL,
  regulator_notice_required BIT NOT NULL DEFAULT 0,
  updated_by BIGINT NULL,
  updated_at DATETIME2 NOT NULL DEFAULT SYSDATETIME(),
  CONSTRAINT fk_bodcod_review_report FOREIGN KEY (report_id)
    REFERENCES bod_cod_deviation_reports(id)
);

CREATE TABLE bod_cod_approval_steps (
  id BIGINT IDENTITY(1,1) PRIMARY KEY,
  report_id BIGINT NOT NULL,
  track VARCHAR(16) NOT NULL, -- CENTRAL | REGIONAL
  step_no INT NOT NULL,
  role_code VARCHAR(32) NOT NULL, -- INSPECTOR | REVIEWER | APPROVER
  role_label NVARCHAR(80) NOT NULL,
  status VARCHAR(24) NOT NULL DEFAULT 'PENDING',
  actor_user_id BIGINT NULL,
  actor_name NVARCHAR(255) NULL,
  actor_position NVARCHAR(255) NULL,
  decision VARCHAR(24) NULL, -- APPROVED | REVISION_REQUESTED
  comment NVARCHAR(MAX) NULL,
  decided_at DATETIME2 NULL,
  revision_no INT NOT NULL DEFAULT 1,
  is_current BIT NOT NULL DEFAULT 1,
  created_at DATETIME2 NOT NULL DEFAULT SYSDATETIME(),
  updated_at DATETIME2 NOT NULL DEFAULT SYSDATETIME(),
  CONSTRAINT fk_bodcod_step_report FOREIGN KEY (report_id)
    REFERENCES bod_cod_deviation_reports(id)
);

CREATE TABLE bod_cod_approval_events (
  id BIGINT IDENTITY(1,1) PRIMARY KEY,
  report_id BIGINT NOT NULL,
  step_id BIGINT NULL,
  event_type VARCHAR(32) NOT NULL,
  actor_user_id BIGINT NOT NULL,
  from_status VARCHAR(32) NULL,
  to_status VARCHAR(32) NULL,
  comment NVARCHAR(MAX) NULL,
  metadata_json NVARCHAR(MAX) NULL,
  created_at DATETIME2 NOT NULL DEFAULT SYSDATETIME()
);
